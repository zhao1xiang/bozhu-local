# 数据兼容性解决方案 - 保证客户数据安全

## 核心原则
- **数据安全第一**：绝不能丢失客户数据
- **平滑升级**：用户无感知升级
- **自动回滚**：失败时自动恢复
- **多重保障**：多种兼容性处理方式

## 推荐方案：渐进式数据库适配

### 方案设计思路
1. **保留原数据库**：不删除，不强制重建
2. **动态字段适配**：运行时处理字段类型差异
3. **智能数据转换**：在应用层处理数据类型
4. **自动修复机制**：检测到问题时自动修复

### 具体实施方案

#### 1. 应用层数据适配（最安全）

```python
# models/patient_adapter.py
class PatientDataAdapter:
    """患者数据适配器 - 处理不同版本的数据格式"""
    
    @staticmethod
    def normalize_vision_data(raw_data):
        """标准化视力数据 - 兼容数字和字符串"""
        vision_fields = ['left_vision', 'right_vision', 'left_vision_corrected', 'right_vision_corrected']
        
        for field in vision_fields:
            if field in raw_data and raw_data[field] is not None:
                # 统一转换为字符串
                if isinstance(raw_data[field], (int, float)):
                    raw_data[field] = str(raw_data[field]) if raw_data[field] != 0 else ""
                elif raw_data[field] is None:
                    raw_data[field] = ""
        
        return raw_data
    
    @staticmethod
    def ensure_required_fields(raw_data):
        """确保必需字段存在"""
        defaults = {
            'medical_card_number': '',
            'remarks': '',
            'is_deleted': False,
            'diagnosis_other': '',
            'drug_type_other': ''
        }
        
        for field, default_value in defaults.items():
            if field not in raw_data:
                raw_data[field] = default_value
        
        return raw_data
```

#### 3. 启动时智能检测和修复

```python
# startup/database_checker.py
class DatabaseCompatibilityChecker:
    """数据库兼容性检查器"""
    
    def __init__(self, db_path):
        self.db_path = db_path
        self.issues = []
        self.fixes_applied = []
    
    def check_and_fix(self):
        """检查并修复数据库兼容性问题"""
        print("🔍 检查数据库兼容性...")
        
        # 1. 备份数据库
        backup_path = self.create_backup()
        print(f"✅ 数据库已备份: {backup_path}")
        
        # 2. 检查字段类型
        vision_issues = self.check_vision_field_types()
        
        # 3. 检查缺失字段
        missing_fields = self.check_missing_fields()
        
        # 4. 应用修复
        if vision_issues or missing_fields:
            print("🔧 发现兼容性问题，正在修复...")
            success = self.apply_fixes(vision_issues, missing_fields)
            
            if success:
                print("✅ 数据库兼容性修复完成")
                return True
            else:
                print("❌ 修复失败，恢复备份...")
                self.restore_backup(backup_path)
                return False
        else:
            print("✅ 数据库兼容性良好")
            return True
    
    def check_vision_field_types(self):
        """检查视力字段类型"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("PRAGMA table_info(patient)")
        columns = cursor.fetchall()
        
        vision_issues = []
        for col in columns:
            col_name, col_type = col[1], col[2]
            if col_name in ['left_vision', 'right_vision', 'left_vision_corrected', 'right_vision_corrected']:
                if col_type == 'REAL':
                    vision_issues.append(col_name)
        
        conn.close()
        return vision_issues
    
    def apply_fixes(self, vision_issues, missing_fields):
        """应用修复"""
        try:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            
            # 修复缺失字段（安全操作）
            for table, field, field_type, default in missing_fields:
                try:
                    cursor.execute(f"ALTER TABLE {table} ADD COLUMN {field} {field_type} DEFAULT '{default}'")
                    print(f"  ✅ 添加字段: {table}.{field}")
                except sqlite3.OperationalError as e:
                    if "duplicate column name" in str(e):
                        print(f"  ℹ️  字段已存在: {table}.{field}")
                    else:
                        raise
            
            # 修复视力字段类型（如果需要）
            if vision_issues:
                print("  🔄 修复视力字段类型...")
                self.fix_vision_fields_safe(cursor)
            
            conn.commit()
            conn.close()
            return True
            
        except Exception as e:
            print(f"  ❌ 修复失败: {e}")
            try:
                conn.rollback()
                conn.close()
            except:
                pass
            return False
    
    def fix_vision_fields_safe(self, cursor):
        """安全地修复视力字段类型"""
        # 使用UPDATE语句将数字转为字符串，而不是重建表
        vision_fields = ['left_vision', 'right_vision', 'left_vision_corrected', 'right_vision_corrected']
        
        for field in vision_fields:
            # 检查字段是否存在
            cursor.execute(f"PRAGMA table_info(patient)")
            columns = [col[1] for col in cursor.fetchall()]
            
            if field in columns:
                # 将NULL值设为空字符串，数字值转为字符串
                cursor.execute(f"""
                    UPDATE patient 
                    SET {field} = CASE 
                        WHEN {field} IS NULL THEN ''
                        WHEN {field} = 0 THEN ''
                        ELSE CAST({field} AS TEXT)
                    END
                """)
                print(f"    ✅ 标准化字段: {field}")
```

```python
# api/error_handler.py
class DatabaseErrorHandler:
    """数据库错误处理器"""
    
    @staticmethod
    def handle_query_error(func):
        """装饰器：处理查询错误"""
        def wrapper(*args, **kwargs):
            try:
                return func(*args, **kwargs)
            except Exception as e:
                if "no such column" in str(e):
                    # 字段不存在，使用兼容查询
                    return DatabaseErrorHandler.fallback_query(*args, **kwargs)
                elif "datatype mismatch" in str(e):
                    # 数据类型不匹配，进行类型转换
                    return DatabaseErrorHandler.type_conversion_query(*args, **kwargs)
                else:
                    raise e
        return wrapper
```
